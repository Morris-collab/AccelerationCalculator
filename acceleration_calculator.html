    <h3>Acceleration Calculator</h3>

    <p>Enter time (s) and velocity (m/s):</p>

    <input type="number" id="timeInput" placeholder="Time (s)">

    <input type="number" id="velocityInput" placeholder="Velocity (m/s)">

    <button onclick="calculateAcceleration()">Calculate</button>

    <p id="result"></p>


    <script>

        function calculateAcceleration() {

            let time = parseFloat(document.getElementById("timeInput").value);

            let velocity = parseFloat(document.getElementById("velocityInput").value);

            if (time > 0) {

                let acceleration = velocity / time;

                document.getElementById("result").innerHTML = "Acceleration: " + acceleration.toFixed(2) + " m/s²";

            } else {

                document.getElementById("result").innerHTML = "Please enter a valid time greater than zero.";

            }

        }

    </script>
